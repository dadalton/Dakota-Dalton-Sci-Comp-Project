1function [ u, SORu ] = helmholtzfun( delta )
  2%Dakota Dalton - 1366027

  7%% Given values and Boundary Conditions
  8
  9% start = 0.15;
 10% step = -0.005;
 11% stop = 0.01;
 12%for i = start:step:stop
 13
 14ax = -pi; ay = ax; %given domain limits, these form a rectangle
 15bx = pi; by = bx;
 16
 17lambda = 1; %given value for lambda
 18%lambda = 0;
 19
 20%delta = 0.015; %step size, same for both x and y
 21
 22x = ax:delta:bx;  %discretizing the domain
 23y = ay:delta:by;
 24
 25gb = (bx-x).^2 .* cos((pi*x)/bx); %boundary conditions for y
 26fb = x .* (bx - x).^2;
 27
 28F = sin(pi * (x - ax)/(bx - ax)) ... %applied force
 29    .* cos((pi/2)*(2*(y - ay)/(by - ay) + 1));
 30%F = 0;
 31
 32%% Gauss-Seidel/Liebmann method
 33
 34u = zeros(length(x)); %initial values of u to be iterated over
 35u(1,:) = gb;          %boundary condition for y (bottom)
 36u(end,:) = fb;        %boundary condition for y (top)
 37u(:,1) = gb(1) + (y-ay)/(by-ay) * (fb(1)-gb(1)); %bc for x (left)
 38
 39iter = 0;                   %used to count number of iterations
 40epsilon = ones(length(x));  %calculating relative change per iteration
 41
 42iterarray = [];
 43epsarray = [];
 44
 45% ucon = u;
 46% loop = 1;
 47% dim = size(u);
 48
 49%while converge1 ~= converge2

 51    %converge2 = converge1;
 52    %delta = i
 53    
 54    while epsilon > 0.01
 55        
 56        uprev = u;              %iteration reference
 57
 58        %sweeping through columns and rows, iterating values
 59        for i = 2:length(y)-1
 60            for j = 2:length(x)-1                
 61                u(j,i) = (u(j+1,i) + u(j-1,i) + u(j,i+1) + u(j,i-1) ...
 62                          - (delta^2) * F(j)) * (1/(4 - delta^2 * lambda));
 63            end
 64
 65            %right side x bc is a Neumann condition (insulated)
 66            u(i,end) = (2*u(i,end-1) + u(i+1,end) + u(i-1,end) - (delta^2)*F(i)) ...
 67                       * (1/(4 - delta^2 * lambda)); 
 68
 69        end
 70
 71        %the largest value change in the matrix will determine the epsilon
 72        epsilon = max(max(abs((u-uprev)./u)));  
 73
 74        iter = iter + 1;        %counting the iterations
 75
 76%         if mod(iter,100) == 0
 77%             disp(iter)
 78%             disp(epsilon)
 79%             iterarray = [iterarray, iter];
 80%             epsarray = [epsarray, epsilon];
 81%         end
 82       
 83    end
 84    %disp(min(min(u)))
 85%     loop = loop + 1;
 86    
 87%     ucon = [ucon, u];
 88%     
 89%     rowsub  = 1:dim(1);
 90%     colsub1 = (loop - 1) * dim(2) + 1 : loop * dim(2);
 91%     colsub2 = (loop - 2) * dim(2) + 1 : (loop - 1) * dim(2);   
 92%         
 93%     if ucon(rowsub,colsub1) == ucon(rowsub, colsub2) ...
 94%        & isfinite(ucon(rowsub, colsub1)) 
 95%         disp(start-loop*abs(step))
 96%         break
 97%     end
 98    
 99%end
100%% Gauss-Seidel with Successive Overrelaxtion (SOR)
101
102SORlambda = 1.2;         %coefficient to expedite convergence
103
104SORu = zeros(length(x)); %initial values of u to be iterated over
105SORu(1,:) = gb;          %boundary condition for y (bottom)
106SORu(end,:) = fb;        %boundary condition for y (top)
107SORu(:,1) = gb(1) + (y-ay)/(by-ay) * (fb(1)-gb(1)); %bc for x (left)
108
109SORiter = 0;                   %used to count number of iterations
110SORepsilon = ones(length(x));  %calculating relative change per iteration
111
112SORiterarray = [];
113SORepsarray = [];
114
115
116
117while SORepsilon > 0.01
118    SORuprev = SORu;              %iteration reference
119    
120    %sweeping through columns and rows, iterating values
121    for i = 2:length(y)-1
122        for j = 2:length(x)-1
123            SORu(j,i) = (SORu(j+1,i) + SORu(j-1,i) + SORu(j,i+1) + SORu(j,i-1) ...
124                - (delta^2) * F(j)) * (1/(4 - delta^2 * lambda));
125            SORu(j,i) = SORlambda * SORu(j,i) + (1-SORlambda) * SORuprev(j,i);
126        end
127        
128        %right side x bc is a Neumann condition (insulated)
129        SORu(i,end) = (2*SORu(i,end-1) + SORu(i+1,end) + SORu(i-1,end) - (delta^2)*F(i)) ...
130            * (1/(4 - delta^2 * lambda)); 
131        SORu(i,end) = SORlambda * SORu(i,end) + (1-SORlambda) * SORuprev(i,end);
132        
133    end
134    %the largest value change in the matrix will determine the epsilon
135    SORepsilon = abs(max(max((SORu-SORuprev)./SORu)));  
136    
137    SORiter = SORiter + 1;                        %counting the iterations
138
